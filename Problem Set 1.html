isPerfectPower <- function(n, power) {
  r <- 2 
  isPerfect <- FALSE
  root <- n^(1/power)
  
  while (n >= r^power) {
    if (n == r^power) {
      isPerfect <- TRUE
      root <- r
      break
    }
    r <- r + 1 
  }
  return(list(isPerfect = isPerfect, root = root))
}
# (b)
findRootPower <- function(n) {
  for (i in 2:50) {
    result <- isPerfectPower(n, power = i)
    if (result$isPerfect == TRUE) {
      return(paste(n, "=", result$root, "^", i))
    }
  }
  return(NULL) 
}
